class Solution {
    public int concatenatedBinary(int n) {
        long result = 0;
        int MOD = 1_000_000_007;
        for(int i=1;i<=n;i++)
        {
            int bits = Integer.toBinaryString(i).length();
            result = result << bits;
            result = result + i;
            result = result % MOD;
        }
        return (int) result;
    }
}
