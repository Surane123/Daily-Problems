import java.util.* ;
import java.io.*; 
import java.util.ArrayList;

public class Solution {
    public static int maxPerimeterTriangle(ArrayList<Integer> arr, int n) {
        // Write your code here.
        Collections.sort(arr,Collections.reverseOrder());
        int k=arr.size();
        for(int i=0;i<k-2;i++)
        {
           int a=arr.get(i);
           int b=arr.get(i+1);
           int c=arr.get(i+2);
           if(a<b+c)
           {
               return a+b+c;
           }
        }
        return 0;
    }
}
